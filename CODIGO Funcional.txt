        // Se o usuÃ¡rio estÃ¡ na pesquisa, mantÃ©m o estado correto
        if (userState[sender]?.status === "after_search") {

                await conn.sendMessage(sender, { text: "ðŸ” *Buscando mais leis relacionadas...*" });
                await searchLegislation(userState[sender].lastQuery, sender, conn, userState[sender].page);
                return;
            } else if (normalizedText === "2") {
                userState[sender] = { status: "waiting_law", page: 1 };
                await conn.sendMessage(sender, { text: "âœï¸ *Digite a nova lei que deseja pesquisar:*" });
                return;
            } else if (normalizedText === "3") {
                delete userState[sender];
                await conn.sendMessage(sender, { text: "ðŸ”š *Atendimento encerrado.* Se precisar novamente, basta enviar uma mensagem!" });
                return;

// ðŸ“œ **Mensagem inicial com saudaÃ§Ã£o dinÃ¢mica**
function getGreeting() {
    const hora = new Date().getHours();
    if (hora >= 5 && hora < 12) return "ðŸŒ… Bom dia!";
    if (hora >= 12 && hora < 18) return "ðŸŒž Boa tarde!";
    return "ðŸŒ™ Boa noite!";
}

async function sendInitialMessage(sender, conn) {

    const message = `${greeting} Eu sou o *Assistente Inteligente ALETO*.\n\n` +

// ðŸ” **Busca legislaÃ§Ã£o na API**
async function searchLegislation(query, sender, conn, page = 1) {
    try {
        const response = await axios.get(`${API_URL}?q=${encodeURIComponent(query)}&page=${page}`);
        const data = response.data;

        if (data.results) {
            let message = "ðŸ“œ *Leis encontradas:*\n\n";
            data.results.forEach((law, index) => {
                message += `*${index + 1}. ${law.title}*\n`;
                message += `ðŸ“„ ${law.snippet}\n`;
                message += `ðŸ”— [Acesse Aqui](${law.link})\n\n`;
            });

            await conn.sendMessage(sender, { text: message });
            await sendOptionsMessage(sender, conn);
        } else {
            await conn.sendMessage(sender, { text: "âš ï¸ NÃ£o encontramos mais leis relacionadas. Tente reformular sua pesquisa." });
            await sendOptionsMessage(sender, conn);
        }
    } catch (error) {
        console.error("âŒ Erro ao buscar legislaÃ§Ã£o:", error.message);
        await conn.sendMessage(sender, { text: "âŒ Erro ao conectar com a API. Tente novamente mais tarde." });
    }
}

require('dotenv').config();

const app = express();

// ðŸ”¹ ConfiguraÃ§Ã£o do Redis
const client = redis.createClient();

// ðŸ”¹ Middleware para logs organizados

// ðŸ”¹ ConfiguraÃ§Ã£o de APIs externas
const GOOGLE_API_KEY = process.env.GOOGLE_API_KEY;
const GOOGLE_CX = process.env.GOOGLE_CX;

if (!GOOGLE_API_KEY || !GOOGLE_CX) {
    console.error("âŒ ERRO: Faltando variÃ¡veis de ambiente (GOOGLE_API_KEY ou GOOGLE_CX). ");
    process.exit(1);
}

const CUSTOM_SEARCH_URL = `https://www.googleapis.com/customsearch/v1?key=${GOOGLE_API_KEY}&cx=${GOOGLE_CX}&q=`;

// ðŸ“Œ DicionÃ¡rio de palavras-chave jurÃ­dicas
const keywords = [
    "lei", "cÃ³digo", "regulamento", "norma", "direito", "portaria",
    "decreto", "constituiÃ§Ã£o", "jurÃ­dico", "justiÃ§a", "processo", "legislaÃ§Ã£o"
];

// ðŸ“Œ Verifica se a busca Ã© especÃ­fica de um paÃ­s ou se assume Brasil
function detectCountry(query) {
    if (/\b(internacional|eua|estados unidos|portugal|espanha|franÃ§a|alemanha)\b/i.test(query)) {
        return query;
    }
    return query + " Brasil";
}

// ðŸ”Ž **1. PrÃ©-processador da Consulta**
function preprocessQuery(query) {
    let words = query.toLowerCase().split(" ");
    words = stopwords.removeStopwords(words, stopwords.pt);
    const containsLegalTerms = words.some(word => keywords.includes(word));
    return { query: words.join(" "), isLegal: containsLegalTerms };
}

// ðŸ” **2. Busca no Google Custom Search com suporte a paginaÃ§Ã£o**
async function searchGoogle(query, start = 1) {
    const googleApiUrl = `${CUSTOM_SEARCH_URL}${encodeURIComponent(query)}&num=${RESULTS_PER_PAGE}&start=${start}`;

    try {
        console.log(`ðŸ” Buscando no Google: ${query} (InÃ­cio: ${start})`);
        const response = await axios.get(googleApiUrl);

        if (!response.data.items || response.data.items.length === 0) {
            console.log("âš ï¸ Nenhum resultado encontrado para essa busca.");
            return [];
        }

        return response.data.items.map(item => ({
            title: item.title,
            link: item.link,
            snippet: item.snippet,
            source: new URL(item.link).hostname
        }));
    } catch (error) {
        console.error("âŒ Erro na busca do Google:", error.message);
        return null;
    }
}

// ðŸ“œ **Endpoint principal para pesquisa de leis com paginaÃ§Ã£o**
app.get(['/search', '/buscar'], async (req, res) => {
    try {
        const query = req.query.q;
        const page = parseInt(req.query.page) || 1;
        const startIndex = (page - 1) * RESULTS_PER_PAGE + 1;

        if (!query) {
            return res.status(400).json({ error: 'O parÃ¢metro "q" Ã© obrigatÃ³rio' });
        }

        console.log(`ðŸš€ ðŸ”¹ [${new Date().toLocaleString()}] Nova pesquisa recebida: "${query}" (PÃ¡gina ${page})`);

        // ðŸ”¹ 1. PrÃ©-processa a pesquisa
        const processedQuery = preprocessQuery(query);
        if (!processedQuery.isLegal) {
            return res.json({ message: "âŒ A pesquisa parece nÃ£o estar relacionada a leis. Tente algo como 'Lei de trÃ¢nsito no Brasil'." });
        }

        // ðŸ”¹ 2. Detecta se a busca deve ser para o Brasil ou outro paÃ­s
        const refinedQuery = detectCountry(processedQuery.query);
        
        const cacheKey = `search-law:${query}:page:${page}`;
        const cachedData = await client.get(cacheKey);
        if (cachedData) {
            console.log(`â™»ï¸ Resultado recuperado do cache para "${query}" (PÃ¡gina ${page})`);
            return res.json(JSON.parse(cachedData));
        }

        // ðŸ”¹ 3. Busca no Google com paginaÃ§Ã£o
        let results = await searchGoogle(refinedQuery, startIndex);

        if (results === null) {
            console.log("âŒ Erro ao buscar no Google, retornando erro para o bot.");
            return res.status(500).json({ error: "Erro ao conectar com o Google. Tente novamente mais tarde." });
        }

        if (results.length > 0) {
            console.log(`âœ… ${results.length} resultados encontrados para "${query}" (PÃ¡gina ${page})`);
            const responsePayload = {
                message: `ðŸ“œ Encontramos ${results.length} leis relacionadas para "${query}" (PÃ¡gina ${page}).`,
                results,
                nextPage: results.length === RESULTS_PER_PAGE ? `/buscar?q=${encodeURIComponent(query)}&page=${page + 1}` : null
            };

            await client.setEx(cacheKey, 3600, JSON.stringify(responsePayload)); // Cache por 1 hora
            return res.json(responsePayload);
        }

        return res.json({ message: "âš ï¸ NÃ£o encontramos mais leis relacionadas. Tente reformular sua pesquisa." });

    } catch (error) {
        console.error('âŒ Erro ao buscar lei:', error);
        res.status(500).json({ error: 'Erro ao processar a solicitaÃ§Ã£o' });
    }
});
